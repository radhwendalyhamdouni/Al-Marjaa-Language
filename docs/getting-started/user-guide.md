## 1. البدء السريع

### التثبيت

```bash
# Linux/macOS
curl -fsSL https://raw.githubusercontent.com/radhwendalyhamdouni/Al-Marjaa-Language/main/install/linux/install.sh | bash

# Windows (PowerShell)
Invoke-WebRequest -Uri "https://raw.githubusercontent.com/radhwendalyhamdouni/Al-Marjaa-Language/main/install/windows/install.ps1" -OutFile install.ps1
.\install.ps1
```

### أول برنامج

أنشئ ملف `hello.mrj`:

```almarjaa
# أول برنامج في لغة المرجع
اطبع("مرحباً بالعالم!")؛
```

شغّل البرنامج:

```bash
almarjaa hello.mrj
```

**النتيجة:**
```
مرحباً بالعالم!
```

## 2. المتغيرات والأنواع

### تعريف المتغيرات

```almarjaa
# متغير نصي
متغير اسم = "أحمد"؛

# متغير رقمي
متغير عمر = 25؛

# متغير عشري
متغير طول = 175.5؛

# متغير منطقي
متجر نشط = صحيح؛

# ثابت (لا يمكن تغييره)
ثابت pi = 3.14159؛
```

### تغيير القيم

```almarjaa
متغير عدد = 10؛
اطبع(عدد)؛        # 10

عدد = 20؛
اطبع(عدد)؛        # 20
```

## 3. العمليات الحسابية

```almarjaa
متغير أ = 10؛
متجر ب = 3؛

# الجمع
اطبع(أ + ب)؛      # 13

# الطرح
اطبع(أ - ب)؛      # 7

# الضرب
اطبع(أ * ب)؛      # 30

# القسمة
اطبع(أ / ب)؛      # 3.33...

# الباقي
اطبع(أ % ب)؛      # 1

# الأس
اطبع(أ ** 2)؛     # 100
```

## 4. الشروط

### شرط بسيط

```almarjaa
متغير درجة = 85؛

إذا درجة >= 90 {
    اطبع("ممتاز")؛
} وإلا إذا درجة >= 75 {
    اطبع("جيد جداً")؛
} وإلا إذا درجة >= 60 {
    اطبع("جيد")؛
} وإلا {
    اطبع("مقبول")؛
}
```

### العوامل المنطقية

```almarjaa
متغير س = 10؛
متجر ص = 20؛

# و
إذا س > 5 و ص > 15 {
    اطبع("كلاهما أكبر")؛
}

# أو
إذا س > 15 أو ص > 15 {
    اطبع("أحدهما أكبر")؛
}

# ليس
إذا ليس س > 15 {
    اطبع("س ليس أكبر من 15")؛
}
```

## 5. الحلقات

### حلقة طالما (while)

```almarjaa
متجر عداد = 0؛

طالما عداد < 5 {
    اطبع("العدد: " + نص(عداد))؛
    عداد = عداد + 1؛
}
```

### حلقة لكل (for each)

```almarjaa
متجر أرقام = [1، 2، 3، 4، 5]؛

لكل رقم في أرقام {
    اطبع("الرقم: " + نص(رقم))؛
}
```

### تحكم الحلقة

```almarjaa
متجر عد = 0؛

طالما عد < 10 {
    إذا عد == 5 {
        توقف؛        # إيقاف الحلقة
    }
    إذا عد == 3 {
        عد = عد + 1؛
        استمر؛       # تخطي التكرار
    }
    اطبع(عد)؛
    عد = عد + 1؛
}
```

## 6. الدوال

### تعريف دالة

```almarjaa
# دالة بسيطة
دالة تحية(اسم) {
    اطبع("مرحباً " + اسم + "!")؛
}

تحية("محمد")؛      # مرحباً محمد!
```

### دالة مع قيمة مرجعة

```almarjaa
دالة مجموع(أ، ب) {
    أعطِ أ + ب؛
}

متجر نتيجة = مجموع(5، 3)؛
اطبع(نتيجة)؛       # 8
```

### دالة مع قيمة افتراضية

```almarjaa
دالة ترحيب(اسم = "زائر") {
    أعطِ "مرحباً " + اسم؛
}

اطبع(ترحيب())؛           # مرحباً زائر
اطبع(ترحيب("علي"))؛      # مرحباً علي
```

## 7. القوائم

### إنشاء قائمة

```almarjaa
# قائمة فارغة
متجر فارغة = []؛

# قائمة بأرقام
متجر أرقام = [1، 2، 3، 4، 5]؛

# قائمة مختلطة
متجر مختلطة = [1، "نص"، صحيح، 3.14]؛
```

### الوصول للعناصر

```almarjaa
متجر أرقام = [10، 20، 30، 40، 50]؛

# العنصر الأول
اطبع(أرقام[0])؛    # 10

# العنصر الأخير
اطبع(أرقام[-1])؛   # 50

# نطاق
اطبع(أرقام[1:3])؛  # [20، 30]
```

### تعديل القائمة

```almarjaa
متجر قائمة = [1، 2، 3]؛

# إضافة عنصر
أضف(قائمة، 4)؛
اطبع(قائمة)؛       # [1، 2، 3، 4]

# إزالة عنصر
أزل(قائمة، 0)؛
اطبع(قائمة)؛       # [2، 3، 4]

# طول القائمة
اطبع(طول(قائمة))؛  # 3
```

## 8. النصوص

### عمليات النصوص

```almarjaa
متجر نص = "مرحباً بالعالم"؛

# الطول
اطبع(طول(نص))؛          # 15

# دمج
اطبع(نص + "!")؛         # مرحباً بالعالم!

# قص
اطبع(جزء(نص، 0، 6))؛    # مرحباً

# بحث
اطبع(موضع(نص، "العالم"))؛  # 8

# استبدال
اطبع(استبدل(نص، "العالم"، "الجميع"))؛  # مرحباً بالجميع
```

### تحويل الأنواع

```almarjaa
# رقم إلى نص
متجر عدد = 42؛
اطبع(نص(عدد))؛          # "42"

# نص إلى رقم
متجر نص_رقم = "123"؛
اطبع(رقم(نص_رقم) + 1)؛  # 124
```

## 9. بناء واجهات GUI

### الطريقة الحالية

```almarjaa
# ملاحظة: هذه الميزة قيد التطوير

# إنشاء هيكل واجهة
متجر واجهة = {
    عنوان: "تطبيقي"؛
    عرض: 400؛
    ارتفاع: 300؛
    عناصر: []؛
}؛

# إضافة عناصر
دالة أضف_زر(نص، إجراء) {
    متجر زر = {
        نوع: "زر"؛
        نص: نص؛
        إجراء: إجراء؛
    }؛
    أضف(واجهة.عناصر، زر)؛
}

أضف_زر("تأكيد"، دالة() { اطبع("تم!")؛ })؛
```

### تصدير الواجهة

```bash
# من سطر الأوامر
almarjaa --export تطبيقي --platform web
```

## 10. تصدير المشاريع

### خطوات التصدير

1. **اكتب البرنامج**

```almarjaa
# myapp.mrj
متجر عنوان = "تطبيقي"؛
اطبع("مرحباً من " + عنوان)؛
```

2. **صدّر للمطلوب**

```bash
# Windows
almarjaa --export myapp --platform windows

# Linux
almarjaa --export myapp --platform linux

# macOS
almarjaa --export myapp --platform macos

# Web
almarjaa --export myapp --platform web
```

3. **النتيجة**

```
build/
└── myapp/
    ├── index.html      # للويب
    ├── myapp.exe       # لويندوز
    ├── myapp           # للينكس/ماك
    └── source.mrj      # المصدر
```

## 11. Vibe Coding

### ما هو؟
Vibe Coding يسمح لك بكتابة نص عربي وتحويله إلى كود.

### كيفية الاستخدام

#### الطريقة 1: من REPL

```bash
almarjaa --repl
```

ثم اكتب:
```
ذكاء
أنشئ متغير س يساوي 10
```

#### الطريقة 2: باستخدام API

```bash
# حمّل نموذج AI أولاً
mkdir -p models
wget -O models/qwen2.5-0.5b-instruct-q4_k_m.gguf \
  'https://huggingface.co/Qwen/Qwen2.5-0.5B-Instruct-GGUF/resolve/main/qwen2.5-0.5b-instruct-q4_k_m.gguf'
```

### أمثلة Vibe Coding

| النص العربي | الكود المُنتج |
|------------|---------------|
| أنشئ متغير س يساوي 10 | `متغير س = 10؛` |
| اطبع مرحبا | `اطبع("مرحبا")؛` |
| إذا س أكبر من 5 اطبع كبير | `إذا س > 5 { اطبع("كبير")؛ }` |

## 12. أمثلة كاملة

### مثال 1: حاسبة

```almarjaa
# === حاسبة لغة المرجع ===

دالة جمع(أ، ب) { أعطِ أ + ب؛ }
دالة طرح(أ، ب) { أعطِ أ - ب؛ }
دالة ضرب(أ، ب) { أعطِ أ * ب؛ }
دالة قسمة(أ، ب) {
    إذا ب == 0 {
        أعطِ "خطأ: القسمة على صفر"؛
    }
    أعطِ أ / ب؛
}

متجر أ = 20؛
متجر ب = 4؛

اطبع("=== الحاسبة ===")؛
اطبع("العددان: " + نص(أ) + " و " + نص(ب))؛
اطبع("الجمع: " + نص(جمع(أ، ب)))؛
اطبع("الطرح: " + نص(طرح(أ، ب)))؛
اطبع("الضرب: " + نص(ضرب(أ، ب)))؛
اطبع("القسمة: " + نص(قسمة(أ، ب)))؛
```

### مثال 2: جدول الضرب

```almarjaa
# === جدول الضرب ===

دالة جدول_ضرب(ن) {
    متجر عد = 1؛
    طالما عد <= 10 {
        اطبع(نص(ن) + " × " + نص(عد) + " = " + نص(ن * عد))؛
        عد = عد + 1؛
    }
}

اطبع("=== جدول الضرب للعدد ٧ ===")؛
جدول_ضرب(7)؛
```

### مثال 3: البحث في قائمة

```almarjaa
# === البحث في قائمة ===

دالة ابحث(قائمة، قيمة) {
    متجر فهرس = 0؛
    طالما فهرس < طول(قائمة) {
        إذا قائمة[فهرس] == قيمة {
            أعطِ فهرس؛
        }
        فهرس = فهرس + 1؛
    }
    أعطِ -1؛  # غير موجود
}

متجر أرقام = [5، 3، 8، 1، 9، 4]؛

اطبع("البحث عن 8: " + نص(ابحث(أرقام، 8)))؛    # 2
اطبع("البحث عن 7: " + نص(ابحث(أرقام، 7)))؛    # -1
```
