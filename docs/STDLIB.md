# المكتبة القياسية - Standard Library

**الإصدار:** 3.3.0  
**آخر تحديث:** 2026

## نظرة عامة

توفر المكتبة القياسية للغة المرجع مجموعة شاملة من الوحدات الأساسية لبناء تطبيقات قوية ومتقدمة:

- **HTTP**: عميل وخادم HTTP متقدم
- **Database**: دعم قواعد البيانات المتعددة
- **Regex**: التعابير النمطية
- **Crypto**: التشفير والأمان

## 1. وحدة HTTP

### 1.1 عميل HTTP

```almarjaa
# إنشاء عميل HTTP
عميل = عميل_جديد()

# طلب GET
استجابة = عميل.احضر("https://api.example.com/data")

# طلب POST
استجابة = عميل.ارسل("https://api.example.com/create", '{"name": "test"}')

# التحقق من النجاح
إذا استجابة.نجاح {
    اطبع(استجابة.نص)
}
```

### 1.2 إعدادات العميل

```almarjaa
# إنشاء عميل بإعدادات مخصصة
إعدادات = إعدادات_عميل()
    .مهلة(30)
    .محاولات(3)
    .وكيل("http://proxy:8080")

عميل = عميل_بإعدادات(إعدادات)
```

### 1.3 خادم HTTP

```almarjaa
# إنشاء خادم
خادم = خادم_جديد()
    .منفذ(8080)
    .مضيف("0.0.0.0")

# إضافة مسار GET
خادم.احضر("/", دالة(طلب) {
    استجابة().نص("مرحباً بالعالم!")
})

# إضافة مسار POST
خادم.ارسل("/api/data", دالة(طلب) {
    بيانات = طلب.جسون()
    استجابة().جسون({"status": "success"})
})

# تشغيل الخادم
خادم.شغل()
```

### 1.4 البرمجيات الوسيطة

```almarjaa
# إضافة سجل الطلبات
خادم.استخدم(سجل_الطلبات)

# إضافة CORS
خادم.استخدم(كورس("*"))

# تحديد معدل الطلبات
خادم.استخدم(حد_المعدل(100, 60))  # 100 طلب في 60 ثانية
```

## 2. وحدة Database

### 2.1 SQLite

```almarjaa
# إنشاء اتصال SQLite
اتصال = اتصال_sqlite("mydb.db")

# إنشاء جدول
اتصال.نفذ("
    أنشئ جدول مستخدمين (
        معرف عدد صحيح مفتاح أساسي,
        اسم نص,
        بريد نص
    )
")

# إدراج بيانات
اتصال.نفذ("
    أدخل في مستخدمين (اسم, بريد)
    قيم ('أحمد', 'ahmed@example.com')
")

# استعلام
نتائج = اتصال.استعلم("اختر * من مستخدمين")
```

### 2.2 MySQL

```almarjaa
# إنشاء اتصال MySQL
إعدادات = إعدادات_اتصال()
    .مضيف("localhost")
    .منفذ(3306)
    .مستخدم("root")
    .كلمة_المرور("password")
    .قاعدة_بيانات("mydb")

اتصال = اتصال_mysql(إعدادات)
```

### 2.3 PostgreSQL

```almarjaa
# إنشاء اتصال PostgreSQL
إعدادات = إعدادات_اتصال()
    .مضيف("localhost")
    .منفذ(5432)
    .مستخدم("postgres")
    .كلمة_المرور("password")
    .قاعدة_بيانات("mydb")
```

### 2.4 MongoDB

```almarjaa
# إنشاء اتصال MongoDB
اتصال = اتصال_mongodb(إعدادات)

# إدراج مستند
مستند = مستند_جديد()

    .أضف("عمر", 25)

اتصال.أدرج("مستخدمين", مستند)

# البحث
نتائج = اتصال.ابحث("مستخدمين", {"عمر": {"$gt": 20}})
```

### 2.5 منشئ الاستعلامات

```almarjaa
# إنشاء استعلام SELECT
استعلام = استعلام()
    .جدول("مستخدمين")
    .اختر(["معرف", "اسم", "بريد"])
    .حيث("عمر", ">", 18)
    .رتب_تنازلي("اسم")
    .حد(10)

نتائج = اتصال.نفذ(استعلام)

# إنشاء استعلام INSERT
استعلام = استعلام()
    .جدول("مستخدمين")
    .أدرج({
        "اسم": "محمد",
        "بريد": "mohammed@example.com",
        "عمر": 30
    })
```

## 3. وحدة Regex

### 3.1 الأنماط الأساسية

```almarjaa
# إنشاء نمط
نمط = نمط("[\u0600-\u06FF]+")  # الأحرف العربية

# التحقق من التطابق
إذا نمط.يطابق("مرحبا") {
    اطبع("النص يحتوي على أحرف عربية")
}

# البحث عن جميع التطابقات
تطابقات = نمط.ابحث_الكل("مرحبا بالعالم 2024")

# استبدال
جديد = نمط.استبدل_الكل("مرحبا بالعالم", "***")
```

### 3.2 المجموعات المسماة

```almarjaa
# نمط مع مجموعات مسماة
نمط = نمط("(?<سنة>\\d{4})-(?<شهر>\\d{2})-(?<يوم>\\d{2})")

# استخراج المجموعات
مجموعات = نمط.مجموعات("2024-03-15")
اطبع(مجموعات.اسم("سنة"))  # 2024
اطبع(مجموعات.اسم("شهر"))  # 03
```

### 3.3 الأنماط الجاهزة

```almarjaa
# البريد الإلكتروني
نمط_بريد = نمط_بريد()

# رقم الهاتف
نمط_هاتف = نمط_هاتف()

# URL
نمط_رابط = نمط_رابط()

# التاريخ
نمط_تاريخ = نمط_تاريخ()

# الأحرف العربية
نمط_عربي = نمط_أحرف_عربية()
```

## 4. وحدة Crypto

### 4.1 دوال الهاش

```almarjaa
# SHA-256
هاش = شا256("مرحبا بالعالم")
اطبع(هاش)  # a7b8c9...

# SHA-512
هاش = شا512("مرحبا بالعالم")

# MD5 (غير آمن!)
هاش = إم_دي5("نص")  # للاستخدام غير الأمني فقط
```

### 4.2 HMAC

```almarjaa
# حساب HMAC
مفتاح = "مفتاح_سري"
رسالة = "مرحبا"
توقيع = hmac_sha256(مفتاح, رسالة)

# التحقق من HMAC
إذا تحقق_hmac(مفتاح, رسالة, توقيع) {
    اطبع("التوقيع صحيح")
}
```

### 4.3 التشفير المتماثل

```almarjaa
# إنشاء مفتاح
مفتاح = مفتاح_عشوائي(32)  # 256 بت

# تشفير
مشفر = شفر("نص سري", مفتاح)

# فك التشفير
أصلي = فك_التشفير(مشفر, مفتاح)
```

### 4.4 التشفير غير المتماثل

```almarjaa
# توليد زوج مفاتيح
زوج = زوج_مفاتيح(2048)

# تشفير بالمفتاح العام
مشفر = شفر_بالعام("نص سري", زوج.عام)

# فك التشفير بالمفتاح الخاص
أصلي = فك_بالخاص(مشفر, زوج.خاص)

# التوقيع الرقمي
توقيع = وقع("رسالة", زوج.خاص)

# التحقق من التوقيع
إذا تحقق("رسالة", توقيع, زوج.عام) {
    اطبع("التوقيع صحيح")
}
```

### 4.5 كلمات المرور

```almarjaa
# تشفير كلمة المرور
هاش = شفر_كلمة_المرور("كلمة_مرور_قوية")

# التحقق من كلمة المرور
إذا تحقق_كلمة_المرور("كلمة_مرور_قوية", هاش) {
    اطبع("كلمة المرور صحيحة")
}

# فحص قوة كلمة المرور
قوة = قوة_كلمة_المرور("MyP@ssw0rd!")
اطبع(قوة.نص_عربي)  # "قوية"
```

### 4.6 JWT Tokens

```almarجaa
# إنشاء JWT
حمولة = حمولة_jwt()
    .أضف("user_id", 123)
    .أضف("role", "admin")
    .صلاحية(3600)  # ساعة واحدة

رمز = أنشئ_jwt(حمولة, "مفتاح_سري")

# التحقق من JWT
إذا تحقق_jwt(رمز, "مفتاح_سري") {
    بيانات = فك_jwt(رمز, "مفتاح_سري")
    اطبع(بيانات.احصل("user_id"))
}
```

### 4.7 توليد عشوائي آمن

```almarjaa
# بايتات عشوائية
بايتات = بايتات_عشوائية(32)

# نص عشوائي
نص = نص_عشوائي(16)

# كلمة مرور عشوائية
كلمة = كلمة_مرور_عشوائية(12)

# رقم عشوائي
رقم = رقم_عشوائي(1, 100)

# رمز تحقق OTP
رمز = رمز_تحقق(6)  # 6 أرقام

# UUID
معرف = uuid_عشوائي()
```

## 5. التوثيق الكامل

### 5.1 أنواع البيانات

| النوع | الوصف |
|-------|-------|
| `HttpRequest` | طلب HTTP الوارد |
| `HttpResponse` | استجابة HTTP |
| `HttpClient` | عميل HTTP |
| `HttpServer` | خادم HTTP |
| `DatabaseConnection` | اتصال بقاعدة البيانات |
| `QueryResult` | نتيجة استعلام |
| `QueryBuilder` | منشئ الاستعلامات |
| `RegexPattern` | نمط التعابير النمطية |

| `Captures` | مجموعات الالتقاط |
| `Key` | مفتاح تشفير |
| `KeyPair` | زوج مفاتيح |
| `HashResult` | نتيجة الهاش |
| `EncryptionResult` | نتيجة التشفير |
| `JwtToken` | رمز JWT |

### 5.2 رموز الأخطاء

| الرمز | المعنى |
|-------|--------|

## 6. أمثلة متقدمة

### 6.1 API Server كامل

```almarjaa
# خادم API كامل مع مصادقة
خادم = خادم_جديد().منفذ(8080)

# برمجيات وسيطة
خادم.استخدم(سجل_الطلبات)
خادم.استخدم(كورس("*"))
خادم.استخدم(حد_المعدل(100, 60))

# مصادقة
خادم.استخدم(دالة(طلب, استجابة) {
    رمز = طلب.رأس("Authorization")
    إذا !تحقق_jwt(رمز, "سر") {
        استجابة.حالة(401).جسون({"خطأ": "مطلوب مصادقة"})
    }
})

# المسارات
خادم.احضر("/api/users", دالة(طلب) {
    مستخدمين = اتصال.استعلم("اختر * من مستخدمين")
    استجابة().جسون(مستخدمين)
})

خادم.ارسل("/api/users", دالة(طلب) {
    بيانات = طلب.جسون()
    هاش = شفر_كلمة_المرور(بيانات.كلمة_المرور)
    اتصال.نفذ("أدخل في مستخدمين (اسم, كلمة_مرور) قيم (?, ?)", [بيانات.اسم, هاش])
    استجابة().حالة(201).جسون({"status": "created"})
})

خادم.شغل()
```

### 6.2 خدمة مصادقة

```almarjaa
# خدمة مصادقة كاملة
دالة تسجيل_الدخول(بريد, كلمة_مرور) {
    مستخدم = اتصال.استعلم_واحد(
        "اختر * من مستخدمين حيث بريد = ?",
        [بريد]
    )
    
    إذا !مستخدم {
        إرجاع {"خطأ": "المستخدم غير موجود"}
    }
    
    إذا !تحقق_كلمة_المرور(كلمة_مرور, مستخدم.كلمة_مرور) {
        إرجاع {"خطأ": "كلمة المرور غير صحيحة"}
    }
    
    حمولة = حمولة_jwt()
        .أضف("user_id", مستخدم.معرف)
        .أضف("role", مستخدم.دور)
        .صلاحية(86400)  # يوم واحد
    
    رمز = أنشئ_jwt(حمولة, "مفتاح_سري")
    
    إرجاع {"رمز": رمز, "مستخدم": مستخدم}
}
```

## 7. الأداء

| العملية | السرعة |
|---------|--------|
| SHA-256 | ~500 MB/s |
| AES-256-GCM | ~1 GB/s |
| RSA-2048 Sign | ~1000 ops/s |
| Regex Match | ~10M chars/s |
| SQLite Query | ~50K ops/s |
